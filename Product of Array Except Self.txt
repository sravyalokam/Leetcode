// PROBLEM OF THE DAY: Product of Array Except Self

// APPROACH:
    // 1. We are given an array nums and need to compute an array where
    //    result[i] is the product of all elements except nums[i].
    // 2. We use the output array to store prefix products.
    // 3. In the first pass, we store the product of all elements to the left of each index.
    // 4. In the second pass (right to left), we multiply the suffix product with the stored prefix.
    // 5. This avoids using division and extra space.

// TIME COMPLEXITY: O(N)

// SPACE COMPLEXITY: O(1) (excluding the output array)

/**
* @param {number[]} nums
* @return {number[]}
*/
var productExceptSelf = function(nums) {
    let n = nums.length;
    let result = new Array(n);

    let prefix = 1;
    for(let i = 0; i < n; i++) {
        result[i] = prefix;
        prefix *= nums[i];
    }

    let suffix = 1;
    for(let i = n - 1; i >= 0; i--) {
        result[i] *= suffix;
        suffix *= nums[i];
    }
    return result;
};

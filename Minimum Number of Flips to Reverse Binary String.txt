// PROBLEM OF THE DAY: Minimum Number of Flips to Reverse Binary String

// PROBLEM LINK: https://leetcode.com/problems/minimum-number-of-flips-to-reverse-binary-string/?envType=problem-list-v2&envId=bit-manipulation

// SOLUTION EXPLANATION: 
    // 1. Convert the integer n to its binary representation.
    // 2. Use two pointers, one from the start and one from the end of the binary string.
    // 3. For every mismatched symmetric pair, increment the count.
    // 4. Each mismatched pair contributes two flips when converting the string to its reversed version,so the final answer is count * 2.

// SPACE COMPLEXITY: O(log N)

// TIME COMPLEXITY: O(log N)

/**
 * @param {number} n
 * @return {number}
 */
var minimumFlips = function(n) {
    let count = 0;
    const binary = n.toString(2);
    let i = 0;
    let j = binary.length - 1;
    while(i < j) {
        if(binary[i] !== binary[j]) {
            count ++;
        }
        i++;
        j--;
    }
    return count * 2;
};

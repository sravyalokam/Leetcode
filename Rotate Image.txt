// PROBLEM OF THE DAY: Rotate Image

// PROBLEM LINK: https://leetcode.com/problems/rotate-image/

// SOLUTION EXPLANATION:
// 1. The problem asks us to rotate a square matrix by 90 degrees in the clockwise direction.
// 2. Instead of modifying the matrix directly while reading it, we first store the original matrix in another matrix called result.
// 3. Then we iterate through every element of the matrix using two loops.
// 4. For every element at position (i, j), we move it to its new rotated position (j, n - 1 - i).
// 5. This mapping correctly places each element where it should be after a 90° clockwise rotation.
// 6. By filling values into the original matrix using this formula, we get the rotated matrix.


class Solution {
public:
    void rotate(vector<vector<int>>& matrix) {
        vector<vector<int>> result(matrix);
        int n = matrix.size();
        int m = matrix[0].size();
        for(int i = 0; i < n; i++) {
            for(int j = 0; j < m; j++) {
                matrix[j][n - 1 - i] = result[i][j];
            }
        }
    }
};

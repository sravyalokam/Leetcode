/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number[]}
 */
var twoSum = function(nums, target) {
    const map = new Map();
    let n = nums.length;
    for(let i = 0; i < n; i++){
        let elem = target - nums[i];
        if(map.has(elem)){
            return [i, map.get(elem)];
        } else {
            map.set(nums[i], i);
        }
    }
    return [];
};

class Solution {
public:
    int lengthOfLastWord(string s) {
        int n = s.size();
        int j=n-1;

        while(j) { 
            if(s[j]==' ') j--; 
            else break;
        }
    
        int cnt=0;
        for(int i=j;i>=0;i--)
        {
            if(s[i]==' '){
               return cnt;
            }
            else{
                cnt++;
            }
        }
        return cnt;
    }
};

/**
 * @param {number} n
 * @return {number}
 */
var bitwiseComplement = function(n) {
    let bit_size = 0;
    if(n === 0) bit_size = 1;
    else bit_size = n.toString(2).length;
    let num = '';
    for(let i = bit_size - 1; i >= 0; i--) {
        if( n & (1 << i)) num += '0';
        else num += '1';
    }
    return parseInt(num, 2);
};

/**
 * @param {string[]} strs
 * @return {number}
 */
var minDeletionSize = function(strs) {
    const n = strs.length;
    const m = strs[0].length;
    let deletions = 0;

    for (let i = 0; i < m; i++) {
        for (let j = 0; j < n - 1; j++) {
            if (strs[j][i] > strs[j + 1][i]) {
                deletions++;
                break; 
            }
        }
    }

    return deletions; 
};

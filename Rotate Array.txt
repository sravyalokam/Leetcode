/**
 * @param {number[]} nums
 * @param {number} k
 * @return {void} Do not return anything, modify nums in-place instead.
 */

const reverse = (nums, l, r) => {
    let temp = [];

    for (let i = r; i >= l; i--) {
        temp.push(nums[i]);
    }

    for (let i = l; i <= r; i++) {
        nums[i] = temp[i - l];
    }
};

var rotate = function(nums, k) {
    const n = nums.length;
    k = k % n;

    reverse(nums, 0, n - 1);
    reverse(nums, 0, k - 1);
    reverse(nums, k, n - 1);
};
